# 1. 在服务器上克隆项目（不包含 .env 文件）
git clone your-repo-url
cd fastapi-zhipu-project

# 2. 在服务器上创建 .env 文件（不要上传到 Git）
echo "ZHIPU_API_KEY=你的真实密钥" > .env
echo "ZHIPU_API_URL=https://open.bigmodel.cn/api/paas/v4/chat/completions" >> .env

# 3. 设置文件权限（可选，提高安全性）
chmod 600 .env

# 4. 安装依赖
pip install -r requirements.txt

# 5. 使用 systemd 或 screen 运行（保持后台运行）
screen -S zhipu-api
uvicorn main:app --host 0.0.0.0 --port 8000
# Ctrl+A, D 分离会话
